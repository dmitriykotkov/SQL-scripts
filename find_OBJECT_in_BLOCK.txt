
select * from dba_tablespaces

select * from dba_data_files

select * from dba_segments where TABLESPACE_NAME in ('TBS_CIF_TRAN_DATA_01','CIF_TRANSACTION2','CIF_A_01','CIF_B_01')

select * from dba_tables where TABLESPACE_NAME in ('TBS_CIF_TRAN_DATA_01','CIF_TRANSACTION2','CIF_A_01','CIF_B_01')
select * from dba_indexes where status<>'VALID' and owner in ('CIF','CIFWAREHOUSE')

select * from dba_users


select /*+ RULE CURSOR_SHARING_EXACT */
  substrb(dbe.owner,1,30) DB_OWNER,
  substrb(dbe.segment_name,1,30 ) OBJ_NAME,
  substrb(dbe.partition_name,1,20 ) PART_NAME,
  substrb(dbe.segment_type,1,3) TYP,
  substrb(dbe.tablespace_name,1,30) TS_NAME
from dba_extents dbe
where dbe.file_id = 8  --< File ID
and 293376             --< Block number 
  between dbe.block_id and dbe.block_id+dbe.blocks-1
