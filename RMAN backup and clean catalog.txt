list backup summary;
allocate channel for maintenance device type sbt parms 'SBT_LIBRARY=oracle.disksbt, ENV=(BACKUP_DIR=/tmp)';
allocate channel for maintenance device type disk;
delete force obsolete;
delete force backup;


#!/bin/bash

export ORACLE_SID=$1
export ORACLE_HOME=$2
export PATH=$ORACLE_HOME/bin:.:/home/oracle:/usr/lib64/qt-3.3/bin:/usr/local/bin:/bin:/usr/bin:/usr/local/sbin:/usr/sbin:/sbin:/home/oracle/bin

cd /cloudfs/dba_scripts

DB_NAME=${ORACLE_SID%%_*}
export BACKUP_DIR=/mnt/netapp/$DB_NAME

rman target / << EOF
CONFIGURE RETENTION POLICY TO RECOVERY WINDOW OF 3 DAYS;
CONFIGURE CONTROLFILE AUTOBACKUP ON;
CONFIGURE CONTROLFILE AUTOBACKUP FORMAT FOR DEVICE TYPE  DISK TO '$BACKUP_DIR/${DB_NAME}_cf_%F';
CONFIGURE BACKUP OPTIMIZATION ON;
CONFIGURE DEVICE TYPE disk PARALLELISM 3;
CONFIGURE DEFAULT DEVICE TYPE TO disk; 
CONFIGURE CHANNEL 1 DEVICE TYPE DISK FORMAT  '$BACKUP_DIR/${DB_NAME}_db_al_%U';
CONFIGURE CHANNEL 2 DEVICE TYPE DISK FORMAT  '$BACKUP_DIR/${DB_NAME}_db_al_%U';
CONFIGURE CHANNEL 3 DEVICE TYPE DISK FORMAT  '$BACKUP_DIR/${DB_NAME}_db_al_%U';
CONFIGURE SNAPSHOT CONTROLFILE NAME TO  '$BACKUP_DIR/${DB_NAME}_snapcf.f';
BACKUP DATABASE PLUS ARCHIVELOG DELETE INPUT;
ALLOCATE CHANNEL FOR MAINTENANCE DEVICE TYPE DISK;
CROSSCHECK BACKUP;
DELETE NOPROMPT EXPIRED BACKUP;
DELETE NOPROMPT OBSOLETE DEVICE TYPE DISK;
CROSSCHECK ARCHIVELOG ALL;
DELETE NOPROMPT EXPIRED ARCHIVELOG ALL;
EOF
